```

```

### 🔴 Bug 2：时间标签严重误导

```
原因：data_fetcher 固定传 years=3 给 akshare，

      Agent 拿到假标签 → 分析逻辑全部基于错误前提
```

### 🟠 逻辑缺陷 1：次新基金风险评分虚低

风险评分 `3.2/10（中等风险）`，但基金只有5.5个月历史。**数据不足本身就是最大风险**，应该更高。

### 🟠 逻辑缺陷 2：投资建议过于激进

综合分析全是"次新基金数据不可信"，但最终仍给出「建议买入」，逻辑自相矛盾。

报告中出现：
- `"研发费用加薪计划"` → 应为"研发费用加计扣除"

# 修复经理姓名解析Bug + 多经理支持

def get_fund_manager_info(manager_name: str) -> Dict:
    """
    获取基金经理信息
    修复：经理姓名可能含换行符或特殊分隔符，需先清洗
    支持：多个经理姓名（逗号/换行分隔），分别查询后合并返回
    """
    if not AKSHARE_AVAILABLE:
        return _mock_manager_info(manager_name)
    
    try:
        # ✅ 修复1：清洗经理姓名，处理换行符、全角逗号、空格等
        # akshare 有时返回 "俞瑶\n要文强" 或 "俞瑶，要文强"
        import re
        raw_names = re.split(r'[\n，,、\s]+', manager_name.strip())
        clean_names = [n.strip() for n in raw_names if n.strip()]
        print(f"DEBUG 解析经理姓名：原始={repr(manager_name)} → 解析={clean_names}")
        
        manager_df = ak.fund_manager_em()
        print(f"DEBUG 经理表列名: {manager_df.columns.tolist()}")
        
        # ✅ 修复2：兼容列名变化
        name_col = next(
            (c for c in ["姓名", "基金经理", "name"] if c in manager_df.columns),
            manager_df.columns[0]
        )
        
        results = []
        for name in clean_names:
            matched = manager_df[manager_df[name_col].str.contains(name, na=False)]
            if not matched.empty:
                row = matched.iloc[0]
                
                # ✅ 修复3：从业时间是天数，转换为年
                exp_raw = row.get("累计从业时间", row.get("从业时间", "0"))
                try:
                    exp_days  = int(str(exp_raw).replace("天", "").strip())
                    exp_years = round(exp_days / 365, 1)
                except (ValueError, TypeError):
                    exp_years = "未知"
                
                results.append({
                    "name":             name,
                    "experience_years": f"{exp_years}年",
                    "managed_funds":    str(row.get("现任基金数量", row.get("管理基金数量", "未知"))),
                    "total_aum":        str(row.get("现任基金总规模(亿元)", row.get("管理总规模", "未知"))) + "亿元",
                    "best_return":      str(row.get("最佳基金回报", "未知")),
                    "data_source":      "akshare"
                })
            else:
                results.append(_mock_manager_info(name))
        
        # ✅ 返回结构：支持多经理
        if len(results) == 1:
            return results[0]
        else:
            return {
                "is_multi_manager": True,
                "managers":         results,
                "manager_count":    len(results),
                "data_source":      "akshare"
            }
    
    except Exception as e:
        print(f"⚠️ akshare 获取经理信息失败，使用模拟数据：{e}")
        return _mock_manager_info(manager_name)

### 修改2：`backend/tools.py` — 修复风险评分逻辑（次新基金惩罚项）

```python
# backend/tools.py
# ============ 修改 tool_calculate_risk_score() ============
# 新增 actual_days 参数，对次新基金增加数据不足惩罚分

@tool
def tool_calculate_risk_score(
    max_drawdown:  float,
    return_rate:   float,
    fund_type:     str,
    actual_days:   int = 1095    # ✅ 新增：实际运行天数，默认3年
) -> str:
    """
    根据基金的最大回撤、收益率、类型和实际运行天数，计算风险评分。
    
    参数：
        max_drawdown:  最大回撤百分比，如 25.3
        return_rate:   收益率百分比，如 45.2
        fund_type:     基金类型，如「股票型」
        actual_days:   基金实际运行天数（用于次新基金惩罚）
    
    返回：综合风险评分（1-10分，10分最高风险）
    """
    try:
        # 1. 回撤评分
        if max_drawdown < 5:      drawdown_score = 1
        elif max_drawdown < 10:   drawdown_score = 2
        elif max_drawdown < 20:   drawdown_score = 4
        elif max_drawdown < 30:   drawdown_score = 6
        elif max_drawdown < 40:   drawdown_score = 8
        else:                     drawdown_score = 10
        
        # 2. 类型基准分
        type_risk_map = {
            "货币型": 1, "债券型": 2, "混合型": 5,
            "股票型": 7, "指数型": 6, "QDII": 8
        }
        type_score = type_risk_map.get(fund_type, 5)
        
        # 3. ✅ 新增：数据不足惩罚（次新基金的核心风险）
        # 🌰 类比：面试一个只工作了3个月的人，
        #         表现再好也不能完全信任，因为样本太少
        if actual_days < 180:
            data_penalty = 3.0    # 不足6个月：+3分（高度不确定）
            data_warning = "⚠️ 严重：运行不足6个月，数据不具统计显著性，风险被严重低估"
        elif actual_days < 365:
            data_penalty = 2.0    # 不足1年：+2分
            data_warning = "⚠️ 警告：运行不足1年，历史数据参考价值有限"
        elif actual_days < 730:
            data_penalty = 1.0    # 不足2年：+1分
            data_warning = "注意：运行不足2年，未经历完整市场周期"
        else:
            data_penalty = 0.0
            data_warning = ""
        
        # 4. 性价比分析
        risk_return_ratio = return_rate / (max_drawdown + 1)
        if risk_return_ratio < 0.5:
            performance_warning = "⚠️ 风险收益比偏低"
        elif risk_return_ratio < 1.5:
            performance_warning = "📊 风险收益比一般"
        else:
            performance_warning = "✅ 风险收益比表面较好（但需结合实际运行时长判断）"
        
        # 5. 综合评分（加入数据惩罚项）
        base_score  = round(drawdown_score * 0.6 + type_score * 0.4, 1)
        final_score = min(10.0, round(base_score + data_penalty, 1))   # ✅ 加入惩罚
        
        # 6. 风险等级（基于最终分）
        if final_score <= 3:
            level  = "低风险 🟢"
            advice = "适合保守型投资者"
        elif final_score <= 5:
            level  = "中等风险 🟡"
            advice = "适合稳健型投资者"
        elif final_score <= 7:
            level  = "中高风险 🟠"
            advice = "适合积极型投资者，需有较强风险承受能力"
        elif final_score <= 8:
            level  = "高风险 🔴"
            advice = "仅适合激进型投资者"
        else:
            level  = "极高风险 🔴🔴"
            advice = "极度谨慎，建议回避"
        
        result = {
            "risk_score":           final_score,
            "base_score":           base_score,
            "data_penalty":         data_penalty,
            "risk_level":           level,
            "data_warning":         data_warning,
            "drawdown_impact":      f"最大回撤{max_drawdown}%，回撤风险分{drawdown_score}/10",
            "type_impact":          f"基金类型[{fund_type}]，类型基准分{type_score}/10",
            "data_insufficiency":   f"运行{actual_days}天，数据不足惩罚+{data_penalty}分",
            "performance_analysis": performance_warning,
            "investment_advice":    advice
        }
        
        return json.dumps(result, ensure_ascii=False, indent=2)
        
    except Exception as e:
        return f"风险评估计算失败：{str(e)}"
```

# backend/graph.py
# ============ 需要修改的部分 ============

# 1. FundAnalysisState 新增字段
class FundAnalysisState(TypedDict):
    fund_code:          str
    fund_name:          str
    user_query:         str
    actual_days:        int          # ✅ 新增：基金实际运行天数
    is_new_fund:        bool         # ✅ 新增：是否次新基金
    market_analysis:    str
    sentiment_analysis: str
    risk_analysis:      str
    final_report:       str
    data_quality:       str
    error_messages:     List[str]
    current_step:       str


# 2. 初始化 state 时补充新字段
def run_fund_analysis(fund_code: str, user_query: str, session_id: str = "default") -> dict:
    initial_state: FundAnalysisState = {
        "fund_code":          fund_code,
        "fund_name":          fund_code,
        "user_query":         user_query,
        "actual_days":        0,         # ✅ 新增，由行情Agent填充
        "is_new_fund":        False,     # ✅ 新增
        "market_analysis":    "",
        "sentiment_analysis": "",
        "risk_analysis":      "",
        "final_report":       "",
        "data_quality":       "",
        "error_messages":     [],
        "current_step":       "开始分析..."
    }
    ...


# 3. run_market_analysis 执行后，从结果提取 actual_days
def run_market_analysis(state: FundAnalysisState) -> FundAnalysisState:
    print(f"\n📊 行情分析师开始工作...")
    
    query = f"""
    请对基金代码 {state['fund_code']} 进行完整的行情分析。
    用户问题：{state['user_query']}
    
    必须按顺序调用：
    1. tool_get_fund_info
    2. tool_get_fund_performance  ← 注意读取返回的 actual_period_label 和 actual_days 字段
    3. tool_get_manager_info（用步骤1返回的经理姓名，传入完整字符串）
    4. tool_compare_fund_ranking
    
    输出报告时必须使用 actual_period_label 作为时间区间描述，禁止写"近3年"。
    """
    
    try:
        config = {"configurable": {"thread_id": f"market_{state['fund_code']}"}}
        result = market_agent.invoke(
            {"messages": [("human", query)]},
            config=config
        )
        analysis = result["messages"][-1].content
        
        # ✅ 从工具调用消息中提取 actual_days（解析 tool message）
        actual_days = 0
        is_new_fund = False
        for msg in result.get("messages", []):
            if hasattr(msg, "content") and isinstance(msg.content, str):
                try:
                    data = json.loads(msg.content)
                    if "actual_days" in data:
                        actual_days = int(data["actual_days"])
                        is_new_fund = data.get("is_new_fund", actual_days < 365)
                        break
                except (json.JSONDecodeError, TypeError):
                    pass
        
        return {
            **state,
            "market_analysis": analysis,
            "actual_days":     actual_days,    # ✅ 传递给后续节点
            "is_new_fund":     is_new_fund,    # ✅ 传递给后续节点
            "current_step":    "行情分析完成 ✅"
        }
    except Exception as e:
        error_msg = f"行情分析失败：{str(e)}"
        return {
            **state,
            "market_analysis": f"行情分析失败：{error_msg}",
            "error_messages":  state.get("error_messages", []) + [error_msg],
            "current_step":    "行情分析失败 ❌"
        }


# 4. run_risk_analysis 传入 actual_days 给风控工具
def run_risk_analysis(state: FundAnalysisState) -> FundAnalysisState:
    print(f"\n⚠️  风险控制官开始工作...")
    
    actual_days = state.get("actual_days", 0)
    
    query = f"""
    请对基金代码 {state['fund_code']} 进行风险评估。
    
    已知该基金实际运行天数为 {actual_days} 天。
    
    步骤：
    1. 调用 tool_get_fund_performance 获取回撤数据
    2. 调用 tool_calculate_risk_score，参数：
       - max_drawdown: 从步骤1获取
       - return_rate:  从步骤1获取
       - fund_type:    从步骤1获取（或使用「混合型」）
       - actual_days:  {actual_days}    ← 必须传入此值，不得省略
    3. 输出风险评估报告
    """
    
    try:
        config = {"configurable": {"thread_id": f"risk_{state['fund_code']}"}}
        result = risk_agent.invoke(
            {"messages": [("human", query)]},
            config=config
        )
        analysis = result["messages"][-1].content
        return {**state, "risk_analysis": analysis, "current_step": "风险评估完成 ✅"}
    except Exception as e:
        error_msg = f"风险分析失败：{str(e)}"
        return {
            **state,
            "risk_analysis":   f"风险分析失败：{error_msg}",
            "error_messages":  state.get("error_messages", []) + [error_msg],
            "current_step":    "风险评估失败 ❌"
        }


# 5. run_report_generation 正确替换 {DATA_QUALITY_NOTICE} 占位符
def run_report_generation(state: FundAnalysisState) -> FundAnalysisState:
    print(f"\n📝 报告撰写员开始汇总...")
    
    # ✅ 构建数据质量说明（替换占位符）
    actual_days = state.get("actual_days", 0)
    is_new_fund = state.get("is_new_fund", False)
    errors      = state.get("error_messages", [])
    
    if is_new_fund:
        data_quality_notice = (
            f"⚠️ **次新基金警告**：该基金仅运行 {actual_days} 天，"
            f"所有历史业绩、排名及风险指标均不具备统计显著性，"
            f"请以官方基金公告为准，本报告分析结论参考价值有限。"
        )
    elif errors:
        data_quality_notice = (
            f"部分数据获取失败（{len(errors)}项），已使用模拟数据填充，"
            f"相关指标仅供参考，请以官方渠道数据为准。"
        )
    else:
        data_quality_notice = "✅ 本报告所有量化数据均来自 akshare 实时接口，数据可信。"
    
    synthesis_query = f"""
    请综合以下三份分析报告，生成最终的基金投研报告。
    
    ===== 行情分析报告 =====
    {state.get('market_analysis', '数据获取失败')}
    
    ===== 舆情分析报告 =====
    {state.get('sentiment_analysis', '数据获取失败')}
    
    ===== 风险评估报告 =====
    {state.get('risk_analysis', '数据获取失败')}
    
    基金代码：{state['fund_code']}
    基金实际运行天数：{actual_days}天（{'次新基金，禁止给出买入评级' if is_new_fund else '数据有效'}）
    用户问题：{state['user_query']}
    
    请将以下内容填入报告七、数据质量说明章节（直接替换 {{DATA_QUALITY_NOTICE}}）：
    {data_quality_notice}
    """
    
    try:
        config = {"configurable": {"thread_id": f"report_{state['fund_code']}"}}
        result = report_agent.invoke(
            {"messages": [("human", synthesis_query)]},
            config=config
        )
        final_report = result["messages"][-1].content
        
        # ✅ 兜底替换（防止 LLM 未替换占位符）
        final_report = final_report.replace("{DATA_QUALITY_NOTICE}", data_quality_notice)
        
        return {
            **state,
            "final_report":  final_report,
            "data_quality":  data_quality_notice,
            "current_step":  "报告生成完成 🎉"
        }
    except Exception as e:
        error_msg = f"报告生成失败：{str(e)}"
        return {
            **state,
            "final_report":  f"报告生成失败，请重试：{error_msg}",
            "error_messages": state.get("error_messages", []) + [error_msg],
            "current_step":  "报告生成失败 ❌"
        }
